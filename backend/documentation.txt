::::::::::: CREATING AN HELPER FUNCION :::::::::::::::::::::::::::::::::::::
1- Create a file in app/Helpers/crypto_helper.php: mkdir -p app/Helpers
2- In app/Helpers/crypto_helper.php, add:
<?php

use App\Models\Crypto;
use Illuminate\Support\Facades\Cache;

if (!function_exists('getCryptoRate')) {
    function getCryptoRate(string $symbol): float
    {
        return Cache::remember("crypto_rate_" . strtoupper($symbol), 300, function () use ($symbol) {
            return Crypto::where('symbol', strtoupper($symbol))->value('usd_rate') ?? 5000;
        });
    }
}


3- In your composer.json, add the file under the autoload > files section:
"autoload": {
    "files": [
        "app/Helpers/crypto_helper.php"
    ]
}
